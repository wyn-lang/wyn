Quick Example

#!/usr/bin/env wyn run import os fn main() { // Method chaining const output = os.command("docker") .arg("ps") .arg("-a") .output() // List comprehensions const numbers = [1, 2, 3, 4, 5] const evens = [x for x in numbers if x % 2 == 0] // Ternary operator const status = evens.len() > 0 ? "found" : "none" print(status) }
